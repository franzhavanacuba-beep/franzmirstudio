<script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>

<script>

/* =========================

   NAVEGACIÓN ENTRE VISTAS

========================= */

function navigateTo(viewId, targetId = null) {


    document.querySelectorAll('.view').forEach(v => {

        v.classList.remove('active')

    })


    const targetView = document.getElementById(viewId + '-view')

    if (!targetView) return


    targetView.classList.add('active')


    window.scrollTo({

        top: 0,

        behavior: "instant"

    })


    setTimeout(() => {

        updateScrollReveals()

    }, 200)


    if (targetId) {

        setTimeout(() => {

            const el = document.getElementById(targetId)

            if (el) {

                el.scrollIntoView({

                    behavior: "smooth",

                    block: "start"

                })

            }

        }, 300)

    }


    if (viewId === "home") {

        setTimeout(() => {

            initMap()

        }, 400)

    }

}



/* =========================

   GALERÍA

========================= */


const galleryContainer = document.getElementById('full-gallery')


function generateGallery() {


    if (!galleryContainer) return


    galleryContainer.innerHTML = ''


    for (let i = 1; i <= 15; i++) {


        const item = document.createElement('div')


        const src = `https://picsum.photos/seed/franzmir${i}/1600/900`


        item.className = 'gallery-item reveal-on-scroll'


        item.innerHTML = `

            <img 

                src="${src}" 

                alt="Obra ${i}" 

                loading="lazy"

            >

        `


        item.querySelector("img").addEventListener("click", () => {

            openLightbox(src)

        })


        galleryContainer.appendChild(item)

    }

}



/* =========================

   LIGHTBOX

========================= */


function openLightbox(src) {

    const box = document.getElementById('lightbox')

    const img = document.getElementById('lightbox-img')


    img.src = src

    box.style.display = "flex"

}


document.addEventListener("keydown", e => {

    if (e.key === "Escape") {

        document.getElementById('lightbox').style.display = "none"

    }

})



/* =========================

   CURSOR PREMIUM

========================= */


const cursor = document.getElementById('cursor')

const follower = document.getElementById('cursor-follower')


document.addEventListener('mousemove', e => {


    cursor.style.transform =

        `translate(${e.clientX}px, ${e.clientY}px)`


    follower.style.transform =

        `translate(${e.clientX - 20}px, ${e.clientY - 20}px)`

})



/* =========================

   NAV SCROLL

========================= */


window.addEventListener('scroll', () => {


    const nav = document.getElementById('main-nav')


    if (window.scrollY > 100) {

        nav.classList.add('nav-scrolled')

    } else {

        nav.classList.remove('nav-scrolled')

    }


    updateScrollReveals()

})



/* =========================

   SCROLL REVEAL

========================= */


function updateScrollReveals() {


    const reveals = document.querySelectorAll('.reveal-on-scroll')


    reveals.forEach(el => {


        const rect = el.getBoundingClientRect()


        if (rect.top < window.innerHeight * 0.85) {

            el.classList.add('visible')

        }

    })

}



/* =========================

   MAPA LEAFLET

========================= */


let mapInstance = null


function initMap() {


    const container = document.getElementById('map')


    if (!container) return


    if (mapInstance) {

        mapInstance.remove()

    }


    const pos = [40.0333, -3.6028]


    mapInstance = L.map('map', {

        scrollWheelZoom: false,

        zoomControl: false,

        dragging: true,

        attributionControl: false

    }).setView(pos, 15)


    L.tileLayer(

        'https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png',

        {

            maxZoom: 20

        }

    ).addTo(mapInstance)


    const icon = L.divIcon({

        html: `

        <div style="

        background:#c5a368;

        width:14px;

        height:14px;

        border-radius:50%;

        box-shadow:0 0 15px rgba(197,163,104,0.8);

        border:2px solid #fff;

        "></div>

        `,

        className: ''

    })


    L.marker(pos, { icon }).addTo(mapInstance)


    setTimeout(() => {

        mapInstance.invalidateSize()

    }, 200)

}



/* =========================

   INIT

========================= */


window.addEventListener("load", () => {


    generateGallery()


    initMap()


    updateScrollReveals()


})

</script>
